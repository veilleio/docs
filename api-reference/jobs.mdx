---
title: "Jobs"
api: "GET https://veille.io/api/v1/jobs/:id"
description: "API endpoint allows you to get a result from async jobs."
---

### Header

<ParamField header="x-api-key" type="string" required={true}>
    This parameter specifies the private key you'll need for Veille.io access.
</ParamField>

### Response

<ResponseField type="object">
    <ResponseField name="type" type="string">
        Allowed values: `CompletedJob` or `QueuedJob`
    </ResponseField>
    <ResponseField name="id" type="string">
        Job ID
    </ResponseField>
    <ResponseField name="endpoint" type="string">
        API endpoint
    </ResponseField>
    <ResponseField name="created_at" type="string">
        Job creation date
    </ResponseField>
    <ResponseField name="started_at" type="string">
        Job start date
    </ResponseField>
    <ResponseField name="duration_ms" type="number">
        Job duration in milliseconds
    </ResponseField>
    <ResponseField name="args" type="object">
        Job arguments
    </ResponseField>
    <ResponseField name="result" type="object">
        Job result
    </ResponseField>
</ResponseField>

<RequestExample>

```bash Example Request
curl --location --request GET 'https://veille.io/api/v1/jobs/018b1e9c-4c91-8b41-7a23-194757ee0a59' \
--header 'Content-Type: application/json' \
--header 'x-api-key: <token>'
```

</RequestExample>

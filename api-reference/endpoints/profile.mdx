---
title: "Linkdin Profile Scraper"
api: "GET https://veille.io/api/v1/scraper/profile"
description: "This endpoint allows you to scrape a LinkedIn profile."
---

### Header

<ParamField header="x-api-key" type="string" required={true}>
    This parameter specifies the private key you'll need for Veille.io access.
</ParamField>

### Parameters

#### Search Query

<ParamField query="query" type="string" required={true}>
    Linkedin profile URL to check : `https://www.linkedin.com/in/username`
</ParamField>

### Response

<ResponseField name="Website" type="object">
    <Expandable title="resume" defaultOpen={false} type="array">
        <ResponseField name="field" type="string">
            Field of expertise
        </ResponseField>
        <ResponseField name="place" type="string">
            Place of work
        </ResponseField>
        <ResponseField name="title" type="string">
            Title
        </ResponseField>
        <ResponseField name="category" type="string">
            Category
        </ResponseField>
        <ResponseField name="date_end" type="string">
            End date
        </ResponseField>
        <ResponseField name="location" type="string">
            Location
        </ResponseField>
        <ResponseField name="place_id" type="string">
            Place ID
        </ResponseField>
        <ResponseField name="full_name" type="string">
            Full name
        </ResponseField>
        <ResponseField name="date_start" type="string">
            Start date
        </ResponseField>
        <ResponseField name="profile_id" type="string">
            Profile ID
        </ResponseField>
        <ResponseField name="description" type="string">
            Description
        </ResponseField>
    </Expandable>
</ResponseField>

<RequestExample>

```bash Example Request
curl --location --request GET 'https://veille.io/api/v1/scraper/profile?query=https://www.linkedin.com/in/williamhgates' \
--header 'Content-Type: application/json' \
--header 'x-api-key: <token>' 
```

</RequestExample>
